# Toast

A message that is displayed temporarily.

<Canvas of={ComponentStories.Default} />

## Usage

You have to add Toaster to your layout like this:

```tsx
import { Toaster } from "mycomponentsui";

export default function RootLayout({ children }) {
  return (
    <html lang='en'>
      <head />
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  );
}
```

Then you can add the toast in all of the children components

```tsx
"use client";

import { Button } from "mycomponentsui";
import { useToast } from "mycomponentsui";

export function ToastSimple() {
  const { toast } = useToast();

  return (
    <Button
      variant='outline'
      onClick={() => {
        toast({
          description: "Your message has been sent.",
        });
      }}
    >
      Show Toast
    </Button>
  );
}
```

## Examples

### With Title

<Canvas of={ComponentStories.WithTitle} />

```tsx
"use client";

import { Button } from "mycomponentsui";
import { useToast } from "mycomponentsui";

export function ToastWithTitle() {
  const { toast } = useToast();

  return (
    <Button
      variant='outline'
      onClick={() => {
        toast({
          title: "Uh oh! Something went wrong.",
          description: "There was a problem with your request.",
        });
      }}
    >
      Show Toast
    </Button>
  );
}
```

### With Action

<Canvas of={ComponentStories.WithAction} />

```tsx

import { Button } from "mycomponentsui";

          action: <ToastAction altText='Try again'>Try again</ToastAction>,
        });
      }}
    >
      Show Toast
    </Button>
  );
}
```
